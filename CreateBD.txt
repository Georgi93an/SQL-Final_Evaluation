CREATE DATABASE VIRTUAL_SCHOOL;

USE VIRTUAL_SCHOOL;

CREATE TABLE MATTER(
CODEMAT CHAR(255) NOT NULL PRIMARY KEY,
WORDING CHAR(255) NOT NULL,
COEF FLOAT NOT NULL
);

CREATE TABLE STUDENT(
NUMSTUD INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
NAME CHAR(255) NOT NULL,
FIRSTNAME CHAR(255) NOT NULL,
DATEOFBIRTH DATE NOT NULL,
STREET CHAR(255) NOT NULL,
POSTALCOD CHAR(255) NOT NULL,
CITY CHAR(255)NOT NULL
);

CREATE TABLE TEST(
NUMTEST INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
TESTDATE DATE NOT NULL,
PLACE CHAR(255),
CODEMAT CHAR(255) NOT NULL,
FOREIGN KEY (CODEMAT) REFERENCES MATTER(CODEMAT)
);

CREATE TABLE NOTATION(
NUMSTUD INT NOT NULL,
NUMTEST INT NOT NULL,
SCORE FLOAT NOT NULL,
PRIMARY KEY (NUMSTUD,NUMTEST),
FOREIGN KEY (NUMSTUD) REFERENCES STUDENT(NUMSTUD),
FOREIGN KEY (NUMTEST)REFERENCES TEST(NUMTEST)
);
